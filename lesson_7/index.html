Объекты и классы

Объявление класса

                
                    <?php
                        class User
                        {
                            public $name;
                            public $email;

                            public function sayHello()
                            {
                                echo "Hello";
                            }
                        }

                        $user = new User();
                        print_r($user);
                        /*
                            User Object
                            (
                                [name] =>
                                [email] =>
                            )
                        */

Создание объекта

                
                    <?php
                        // $user1 содержит экземпляр класса User
                        // instance of class User
                        $user1 = new User();
                        $user2 = new User();

                        var_dump($user1 === $user2); // false

Свойства класса

                
                    <?php
                        class User
                        {
                            // свойства класса
                            public $name;
                            public $email;

                            public function sayHello()
                            {
                                echo "Hello";
                            }
                        }

                        $user = new User();
                        // присвоение значений свойствам
                        $user->name = 'Afanasy';
                        $user->email = 'afa@nas.com';

                        print_r($user);
                        /*
                            User Object
                            (
                                [name] => Afanasy
                                [email] => afa@nas.com
                            )
                        */

Доступ к методам

                        class User

$this

                
                    <?php
                        class User
                        {
                            public $firstName;
                            public $lastName;

                            public function sayName()
                            {
                                echo $this->firstName . ' ' $this->lastName;
                            }
                        }

                        $user = new User();
                        $user->firstName = 'John';
                        $user->lastName = 'Snow';

                        $user->sayName(); // John Snow

Инициализация

                
                    <?php
                        class User
                        {
                            public $firstName;
                            public $lastName;

                            public function __construct($firstName, $lastName)
                            {
                                $this->firstName = $firstName;
                                $this->lastName = $lastName;
                            }

                            public function __toString()
                            {
                                echo $this->firstName . ' ' $this->lastName;
                            }
                        }

                        $user = new User('John', 'Snow');

Модификаторы доступа

Наследование

                
                    <?php
                        class Father
                        {
                            public function say()
                            {
                                echo "I'm your father";
                            }
                        }

                        class Son extends Father
                        {
                            public function say()
                            {
                                echo "Nooooo!";
                            }
                        }

                        $father = new Father();
                        $son = new Son();

                        $father->say(); // I'm your father
                        $son->say(); // Nooooo!
